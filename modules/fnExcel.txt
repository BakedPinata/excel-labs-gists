XLOOKUP = LAMBDA(lookup_value, lookup_array, return_array, [if_not_found], [match_mode], [search_mode],
    LET(
        X, XLOOKUP(lookup_value, lookup_array, return_array, if_not_found, match_mode, search_mode),
        IF(LEN(X)=0,"",X)
    )
);
